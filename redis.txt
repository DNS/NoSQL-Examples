# install redis, as root
su
curl -fsSL https://packages.redis.io/gpg | gpg --dearmor -o /usr/share/keyrings/redis-archive-keyring.gpg
echo "deb [signed-by=/usr/share/keyrings/redis-archive-keyring.gpg] https://packages.redis.io/deb $(lsb_release -cs) main" | tee /etc/apt/sources.list.d/redis.list
apt update
apt -y install redis 

# install redis gui, as root
su
nodejs npm
npm install -g redis-commander
exit
# run as user
redis-commander

==================================

# run server
redis-server

# run cli
# Syntax
redis-cli -u redis://host:port
redis-cli -u redis://username:password@host:port

# Examples
redis-cli
redis-cli -u redis://localhost:6379
redis-cli -u redis://myuser:mypassword@localhost:6379

# If you run Redis through Docker
docker exec -it <container-id-or-name> redis-cli

# Test connection
PING

CLEAR

# Set and get values
SET myKey "Hello"
GET myKey

MGET myKey nonExistentKey
INCR myCounter
KEYS my*
EXISTS myKey

# expire in 3 seconds #1
SET myKey "Hello"
EXPIRE myKey 3
GET myKey

# expire in 3 seconds #2
SETEX myKey 3 "value"
TTL myKey

# not expire (this is the default)
SET myKey "Hello"
PERSIST myKey
TTL myKey

# find key
SCAN 0 MATCH myKe* COUNT 2

# Delete key
DEL myKey

INFO server
INFO keyspace

# List: Ordered collections of strings

# Push/Pop/Get elements
LPUSH mylist "first"
RPUSH mylist "last"
LPOP mylist
RPOP mylist
LRANGE mylist 0 -1
LINDEX mylist 0
LLEN mylist

--------------
# Sets: Unordered collections of unique strings

# Add members
SADD myset "member1"
SADD myset "member2" "member3"

# Check membership
SISMEMBER myset "member1"

# Get members
SMEMBERS myset

# Set operations
SUNION set1 set2
SINTER set1 set2
SDIFF set1 set2

--------------

# Hashes: key-value pairs
HSET user:1 name "John"
HSET user:1 age 30
HGET user:1 name
HGETALL user:1
HMSET user:2 name "Jane" age 25 city "NYC"

# multi hashes
HSET myHash a:b c
HGET myhash a:b
HGETALL myHash
